export function SuccessAlert() {
  return (
    <div className="alert alert-success alert-dismissible fade show" role="alert">
      <i className="bi bi-check alert-icon"></i>
      <div>
        <strong className="alert-title">Success</strong> Your action was completed successfully.
        <div className="alert-footer">
          <button type="button" className="btn btn-subtle">Learn More</button>

        </div>

    </div>
  );
}

# Alerts

Provide contextual feedback with SDGA UI alerts. Our alert system includes multiple styles, icons, and interactive footer actions for user guidance.

## Standard Alerts

Our alerts support contextual variants with icons and action footers to guide user interactions.
